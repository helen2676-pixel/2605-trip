async function initFb(cfg) {
  try {
    if (typeof firebase === 'undefined') { console.warn("Firebase SDK 미로드"); return false; }
    
    // 앱 초기화 로직 (충돌 방지)
    if (firebase.apps.length === 0) {
      firebase.initializeApp(cfg);
    } else if (firebase.app().options.projectId !== cfg.projectId) {
      await firebase.app().delete();
      firebase.initializeApp(cfg);
    }
    
    db = firebase.firestore();
    const ref = db.collection("kyushu2026").doc("shared");

    if (fbUnsub) fbUnsub();

    fbUnsub = ref.onSnapshot(snap => {
      if (!snap.exists) return;

      const data = snap.data();
      if (!data || !data.itinerary) return;

      const remote = JSON.stringify(data.itinerary);
      const local = JSON.stringify(itinerary);

      if (data.updatedBy === CLIENT_ID) return;

      if (remote !== local) {
        itinerary = data.itinerary;
        localStorage.setItem(STORAGE_KEY, JSON.stringify(itinerary));
        
        // 사용자가 드래그 중일 때는 렌더링을 멈췄다가, 마우스를 뗄 때 반영합니다. (버그 방지)
        if (document.body.classList.contains("dragging-active")) {
          showToast("🔄 업데이트 대기 중 (드래그 종료 후 반영)");
          window.pendingRender = true;
        } else {
          render();
          showToast("🔄 공유 일정 업데이트됨");
        }
      }
    }, err => {
      console.error("Firestore sync error:", err);
    });

    const firstSnap = await ref.get();
    if (firstSnap.exists) {
      const serverData = firstSnap.data();
      if (serverData && serverData.itinerary) {
        itinerary = serverData.itinerary;
        localStorage.setItem(STORAGE_KEY, JSON.stringify(itinerary));
      }
    } else {
      await ref.set({
        itinerary,
        updatedAt: firebase.firestore.FieldValue.serverTimestamp(),
        updatedBy: CLIENT_ID
      });
    }

    fbReady = true;
    document.getElementById("fb-banner").classList.add("ok");
    document.getElementById("fb-status").textContent = "✅ 연결됨";
    document.getElementById("sync-dot").classList.add("on");
    document.getElementById("sync-label").textContent = "온라인";
    window.lastFbError = null;

    render();
    return true;
  } catch(e) {
    console.error("Firebase Init Error:", e);
    window.lastFbError = e.message; // 알림창 출력을 위해 저장
    return false;
  }
}

      const rain = d.daily.precipitation_sum[idx];
      weatherData[date] = {
        icon: getIcon(code), rainy: isRainy(code),
        max: Math.round(d.daily.temperature_2m_max[idx]),
        min: Math.round(d.daily.temperature_2m_min[idx]),

            time: t.slice(11,16),
            icon: getIcon(d.hourly.weathercode[hi]),
            temp: Math.round(d.hourly.temperature_2m[hi]),
            prob: d.hourly.precipitation_probability[hi],
          };
        })
      };
    });

    const todayStr = new Date().toISOString().slice(0,10);
    const pastMessages = ["✨완료","🎉추억","🌟끝","💫다녀옴","🗓기록"];

    const dateShorts = ["5/23","5/24","5/25","5/26","5/27"];
    bar.innerHTML = tripDates.map((date, i) => {
      const isPast = date < todayStr;
      const w = weatherData[date];
      if (!w) {
        if (isPast) return `<div class="weather-day past" id="wday-${date}"><div class="wd-date">${dayLabels[i]}</div><div class="wd-date">${dateShorts[i]}</div><div class="wd-icon" style="font-size:0.8rem">${pastMessages[i]}</div></div>`;
        return `<div class="weather-day" id="wday-${date}"><div class="wd-date">${dayLabels[i]}</div><div class="wd-date">${dateShorts[i]}</div><div class="wd-icon">🌡</div></div>`;
      }

let dragState = null;

document.addEventListener("selectstart", e => {
  if (document.body.classList.contains("dragging-active")) e.preventDefault();
});

function clearDropIndicators() {
  document.querySelectorAll(".insert-above,.insert-below,.drag-over")
    .forEach(el => el.classList.remove("insert-above","insert-below","drag-over"));
}

function getInsertTarget(e, sectionId, dayId) {
  const container = document.getElementById(`mc-${dayId}-${sectionId}`);
  if (!container) return { container, insertBeforeId: null };
  const cards = [...container.querySelectorAll(".sched-card:not(.dragging)")];
  for (const card of cards) {
    const rect = card.getBoundingClientRect();
    const midY = rect.top + rect.height / 2;
    if (e.clientY < midY) return { container, insertBeforeId: card.id.replace("card-",""), card };
  }
  return { container, insertBeforeId: null };
}

function onDragStart(e, dayId, sectionId, cardId) {
  dragState = { dayId, sectionId, cardId };
  e.dataTransfer.effectAllowed = "move";
  document.body.classList.add("dragging-active");
  setTimeout(() => document.getElementById("card-" + cardId)?.classList.add("dragging"), 0);
}
function onDragEnd(e) {
  stopEdgeScroll();
  document.body.classList.remove("dragging-active");
  document.querySelectorAll(".dragging").forEach(el => el.classList.remove("dragging"));
  clearDropIndicators();
  dragState = null;
  checkPendingRender(); // 동기화 지연 렌더링 호출
}
function onDragOver(e, dayId, sectionId) {
  e.preventDefault();
  startEdgeScroll(e.clientY);
  clearDropIndicators();
  const { container, insertBeforeId, card } = getInsertTarget(e, sectionId, dayId);
  if (container) container.classList.add("drag-over");
  if (card) card.classList.add("insert-above");
  else if (container) {
    const last = container.querySelector(".sched-card:not(.dragging):last-of-type");
    if (last) last.classList.add("insert-below");
  }
  if (dragState) dragState.insertBeforeId = insertBeforeId;
  if (dragState) dragState.toSectionId = sectionId;
  if (dragState) dragState.toDayId = dayId;
}
function onDragLeave(e) {
  if (e.currentTarget.contains(e.relatedTarget)) return;
  clearDropIndicators();
}
function onDrop(e, toDayId, toSectionId) {
  stopEdgeScroll();
  e.preventDefault();
  if (!dragState) return;
  const { dayId: fromDayId, sectionId: fromSectionId, cardId, insertBeforeId } = dragState;
  moveCard(fromDayId, fromSectionId, cardId, toDayId, toSectionId, insertBeforeId);
  clearDropIndicators();
  dragState = null;
}

// ═══════════════════════════════════════════
//  EDGE AUTO SCROLL
// ═══════════════════════════════════════════
let edgeScrollVelocity = 0;
let edgeScrollFrame = null;

function startEdgeScroll(pointerY) {
  const zone = 120;
  const vh = window.innerHeight;
  if (pointerY < zone) {
    edgeScrollVelocity = -Math.max(4, ((zone - pointerY) / zone) * 18);
  } else if (pointerY > vh - zone) {
    edgeScrollVelocity = Math.max(4, ((pointerY - (vh - zone)) / zone) * 18);
  } else {
    stopEdgeScroll();
    return;
  }
  if (!edgeScrollFrame) {
    const step = () => { window.scrollBy(0, edgeScrollVelocity); edgeScrollFrame = requestAnimationFrame(step); };
    edgeScrollFrame = requestAnimationFrame(step);
  }
}
function stopEdgeScroll() {
  edgeScrollVelocity = 0;
  if (edgeScrollFrame) { cancelAnimationFrame(edgeScrollFrame); edgeScrollFrame = null; }
}

//  TOUCH DRAG (Mobile)
// ═══════════════════════════════════════════
let touchDrag = null;
let touchClone = null;

function parseMcId(mcId) {
  const raw = mcId.slice(3);
  const firstDash = raw.indexOf("-");
  if (firstDash === -1) return null;
  const dayId = raw.slice(0, firstDash);
  const sectionId = raw.slice(firstDash + 1);
  if (!dayId || !sectionId) return null;
  return { dayId, sectionId };
}

function getTouchInsertTarget(clientX, clientY, mc) {
  const cards = [...mc.querySelectorAll(".sched-card")].filter(el => el.id !== "card-" + (touchDrag && touchDrag.cardId));
  for (const card of cards) {
    const rect = card.getBoundingClientRect();
    if (clientY < rect.top + rect.height / 2) return card.id.replace("card-", "");
  }
  return null;
}

function onTouchStart(e, dayId, sectionId, cardId) {
  if (e.target.closest("button, a, input")) return;
  const touch = e.touches[0];
  const card = document.getElementById("card-" + cardId);
  const rect = card.getBoundingClientRect();
  touchDrag = { dayId, sectionId, cardId, startX: touch.clientX, startY: touch.clientY, active: false, insertBeforeId: null };

  touchDrag.timer = setTimeout(() => {
    touchDrag.active = true;
    document.body.classList.add("dragging-active");
    touchClone = card.cloneNode(true);
    touchClone.style.cssText = `position:fixed;left:${rect.left}px;top:${rect.top}px;width:${rect.width}px;opacity:0.88;z-index:9999;pointer-events:none;transform:scale(1.04);box-shadow:0 12px 32px rgba(0,0,0,0.28);border-radius:12px;`;
    document.body.appendChild(touchClone);
    card.style.opacity = "0.25";
    navigator.vibrate && navigator.vibrate(50);
  }, 320);
}

function onTouchMove(e) {
  if (!touchDrag) return;
  const touch = e.touches[0];
  const dx = Math.abs(touch.clientX - touchDrag.startX);
  const dy = Math.abs(touch.clientY - touchDrag.startY);
  if ((dx > 8 || dy > 8) && !touchDrag.active) {
    clearTimeout(touchDrag.timer);
    touchDrag = null;
    return;
  }
  if (!touchDrag || !touchDrag.active || !touchClone) return;
  e.preventDefault();
  startEdgeScroll(touch.clientY);
  const w = touchClone.offsetWidth;
  const h = touchClone.offsetHeight;
  touchClone.style.left = (touch.clientX - w / 2) + "px";
  touchClone.style.top  = (touch.clientY - h / 2) + "px";

  clearDropIndicators();
  touchClone.style.display = "none";
  const under = document.elementFromPoint(touch.clientX, touch.clientY);
  touchClone.style.display = "";
  const mc = under?.closest("[id^='mc-']");
  if (mc) {
    mc.classList.add("drag-over");
    const insertBeforeId = getTouchInsertTarget(touch.clientX, touch.clientY, mc);
    touchDrag.insertBeforeId = insertBeforeId;
    touchDrag.dropMcId = mc.id;
    if (insertBeforeId) {
      document.getElementById("card-" + insertBeforeId)?.classList.add("insert-above");
    } else {
      const cards = mc.querySelectorAll(".sched-card");
      if (cards.length > 0) cards[cards.length - 1].classList.add("insert-below");
    }
  } else {
    touchDrag.dropMcId = null;
  }
}

function onTouchEnd(e) {
  if (!touchDrag) return;
  clearTimeout(touchDrag.timer);

  if (touchDrag.active) {
    if (touchClone) { touchClone.remove(); touchClone = null; }
    const cardEl = document.getElementById("card-" + touchDrag.cardId);
    if (cardEl) cardEl.style.opacity = "";
    stopEdgeScroll();
    document.body.classList.remove("dragging-active");
    clearDropIndicators();

    if (touchDrag.dropMcId) {
      const parsed = parseMcId(touchDrag.dropMcId);
      if (parsed) {
        moveCard(touchDrag.dayId, touchDrag.sectionId, touchDrag.cardId,
                 parsed.dayId, parsed.sectionId, touchDrag.insertBeforeId);
      }
    }
  }
  touchDrag = null;
  checkPendingRender(); // 동기화 지연 렌더링 호출
}

//  MOVE CARD & THEME LOGIC
// ═══════════════════════════════════════════
function moveCard(fromDayId, fromSectionId, cardId, toDayId, toSectionId, insertBeforeId) {
  const fromDay = findDay(fromDayId);
  const toDay   = findDay(toDayId);
  if (!fromDay || !toDay) return;
  const fromSec = fromDay.sections[fromSectionId];
  const toSec   = toDay.sections[toSectionId];
  if (!fromSec || toSec === undefined) return;
  const idx = fromSec.findIndex(c => c.id === cardId);
  if (idx === -1) return;
  const [card] = fromSec.splice(idx, 1);

  if (insertBeforeId) {
    const insertIdx = toSec.findIndex(c => c.id === insertBeforeId);
    if (insertIdx !== -1) toSec.splice(insertIdx, 0, card);
    else toSec.push(card);
  } else {
    toSec.push(card);
  }

  updateDayTheme(fromDayId);
  updateDayTheme(toDayId);
  save(); render();
  showToast(`📦 ${card.title} 이동 완료`);
}

function updateDayTheme(dayId) {
  const day = findDay(dayId);
  if (!day) return;

  const allCards = Object.values(day.sections).flat();

  // 1. 관광지 우선 탐색
  let highlights = allCards
    .filter(c => !c.isGasCard && c.title && c.cat === "attraction")
    .map(c => c.title.trim());

  // 2. 관광지가 없으면 식당 탐색
  if (highlights.length === 0) {
    highlights = allCards
      .filter(c => !c.isGasCard && c.title && c.cat === "restaurant")
      .map(c => c.title.trim());
  }

  highlights = highlights
    .filter(title => title.length > 1 && !title.includes("체크인") && !title.includes("복귀") && !title.includes("출발 준비"))
    .map(title => {
      // '점심 · 스케상우동' 같은 타이틀에서 불필요한 접두어 제거하고 명소/가게 이름만 깔끔하게 유지
      if (title.includes(" · ")) {
        return title.split(" · ")[1].trim();
      }
      return title;
    })
    .filter((title, idx, arr) => arr.indexOf(title) === idx)
    .slice(0, 2);

  day.theme = highlights.length > 0
    ? highlights.join(" · ") 
    : "자유 일정"; // 글자 깎이고 줄임표 나오는 이슈 수정
}

function saveEdit() {

  const { dayId, sectionId, cardId } = editTarget;
  const card = findCard(dayId, sectionId, cardId);
  if (!card) return;
  card.title    = document.getElementById("edit-title").value.trim() || card.title;
  card.cat      = document.getElementById("edit-cat").value;
  card.time     = document.getElementById("edit-time").value.trim();
  card.desc     = document.getElementById("edit-desc").value.trim();
  card.mapQuery = document.getElementById("edit-map").value.trim();

  const newDayId  = document.getElementById("edit-day").value;
  const newSecId  = document.getElementById("edit-meal").value;
  if (newDayId !== dayId || newSecId !== sectionId) {
    moveCard(dayId, sectionId, cardId, newDayId, newSecId);
  } else {
    updateDayTheme(dayId); // 테마 반영 강제 갱신
    save(); render();
  }

}
function deleteCard() {
  if (!editTarget) return;
  const { dayId, sectionId, cardId } = editTarget;
  const day = findDay(dayId);
  if (!day) return;
  const sec = day.sections[sectionId];
  const idx = sec.findIndex(c => c.id === cardId);
  if (idx !== -1) { 
    sec.splice(idx, 1); 
    updateDayTheme(dayId);
    save(); render(); 
  }
  closeEditModal();
}
